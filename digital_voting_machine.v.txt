`timescale 1ns/1ps

module digital_voting_machine(

input clk,
input reset,

input vote1,
input vote2,
input vote3,
input vote4,

input result,

output reg [7:0] candidate1,
output reg [7:0] candidate2,
output reg [7:0] candidate3,
output reg [7:0] candidate4

);

always @(posedge clk or posedge reset)
begin

    if(reset)
    begin

        candidate1 <= 0;
        candidate2 <= 0;
        candidate3 <= 0;
        candidate4 <= 0;

    end

    else
    begin

        if(!result)
        begin

            if(vote1)
                candidate1 <= candidate1 + 1;

            else if(vote2)
                candidate2 <= candidate2 + 1;

            else if(vote3)
                candidate3 <= candidate3 + 1;

            else if(vote4)
                candidate4 <= candidate4 + 1;

        end

    end

end

endmodule
